\begin{enumerate}
\item Directly calculate $\displaystyle\sum\limits_{d|12}\phi(d)$ and verify that you obtain 12 as your
answer.
\begin{align*}
\sum_{d|12}\phi(d) &= \phi(1) + \phi(2) + \phi(3) + \phi(4) + \phi(6) + \phi(12)\\
&= 1 + 1 + 2 + 2 + 2 + 4\\
&= 12
\end{align*}
\end{enumerate}
